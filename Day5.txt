# find the maximum and minimum elements :
# que:write a function to find the maximum and minimum elements in an array
# logic: iterate through the array updating the minimum and maximum as you go 
# sample input [5 3 9 2 8]
# expected output : maximum:9,minimum:2 
def find_max_min(arr):
    maximum = arr[0]
    minimum = arr[0]

    for i in arr:
        if i > maximum:
            maximum = i
        if i < minimum:
            minimum = i

    return maximum, minimum


arr = [5, 3, 9, 2, 8]

max_val, min_val = find_max_min(arr)

print("Maximum:", max_val)
print("Minimum:", min_val)



def find_max_min(arr):

    max_val = arr[0]     # assume first element as maximum
    min_val = arr[0]     # assume first element as minimum

    for i in range(1, len(arr)):   # loop from 2nd element
        if arr[i] > max_val:
            max_val = arr[i]

        if arr[i] < min_val:
            min_val = arr[i]

    print("Maximum:", max_val)

# rremove duplicates from unsorted array:
# que:write a function to remove duplicates from an unsorted array 
# sample input:[3,2,3,1,2,4]
# expected output:[3,2,1,4]
def remove_duplicates(arr):

    result = []

    for i in arr:
        if i not in result:
            result.append(i)

    return result


arr = [3, 2, 3, 1, 2, 4]

output = remove_duplicates(arr)

print(output)
